# 拼豆解压小游戏 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 在桌面 `拼豆/` 文件夹里做一个零依赖的「照图拼豆」解压小游戏，配一座 Node 桥与 Claude Code 联动，代码跑完时多方式提醒并把注意力拉回终端。

**Architecture:** 自包含 `index.html`（Canvas 游戏 + WebAudio + 存档 + CC 面板 + 防沉迷）由零依赖 `server.js`（Node 内置 http）serve；CC hooks 用 curl 把事件 POST 到 `server.js`，后者按 `session_id` 维护会话表并经 SSE 推给页面。

**Tech Stack:** Vanilla JS、Canvas 2D、WebAudio、Server-Sent Events、Node 内置 `http`（零 npm 依赖）、`node:test`（仅测 server 逻辑）。

参考 spec：`docs/2026-06-07-拼豆解压小游戏-design.md`

```js
'use strict';
function emptyState() { return { sessions: {} }; }
function project(cwd) { return (cwd || '').split('/').filter(Boolean).pop() || '未知项目'; }

// 纯函数：返回新 state（浅拷贝），不修改入参
function reduce(state, evt, now) {
  const id = evt.session_id || 'unknown';
  const prev = state.sessions[id] || {
    id, project: project(evt.cwd), state: 'running',
    toolCalls: 0, lastTool: null, startedAt: now, lastEventAt: now, completedAt: null,
  };
  const s = { ...prev, project: project(evt.cwd) || prev.project, lastEventAt: now };
  switch (evt.hook_event_name) {
    case 'SessionStart':
      Object.assign(s, { state: 'running', toolCalls: 0, startedAt: now, completedAt: null });
      break;
    case 'PostToolUse':
    case 'SubagentStop':
      s.state = 'running'; if (evt.hook_event_name === 'PostToolUse') { s.toolCalls = prev.toolCalls + 1; s.lastTool = evt.tool_name || prev.lastTool; }
      break;
    case 'Notification': s.state = 'waiting'; break;
    case 'Stop': s.state = 'done'; s.completedAt = now; break;
    default: break;
  }
  return { ...state, sessions: { ...state.sessions, [id]: s } };
}

const IDLE_MS = 30 * 60 * 1000;
function pruned(state, now) {
  const out = {};
  for (const [id, s] of Object.entries(state.sessions)) if (now - s.lastEventAt < IDLE_MS) out[id] = s;
  return { ...state, sessions: out };
}
module.exports = { emptyState, reduce, pruned, project };
```

## Task 2: server.js — serve 页面 + /hook + /events(SSE)

**Files:**
- Create: `拼豆/server.js`

- [ ] **Step 1: 实现 server.js**

```js
'use strict';
const http = require('http');
const fs = require('fs');
const path = require('path');
const { emptyState, reduce, pruned } = require('./sessions.js');

const PORT = Number(process.env.PINBEADS_PORT || 4500);
let state = emptyState();
const clients = new Set(); // SSE 响应对象

function broadcast() {
  const data = `data: ${JSON.stringify(state.sessions)}\n\n`;
  for (const res of clients) { try { res.write(data); } catch (_) {} }
}

const server = http.createServer((req, res) => {
  const url = new URL(req.url, `http://localhost:${PORT}`);
  if (req.method === 'GET' && (url.pathname === '/' || url.pathname === '/index.html')) {
    const html = fs.readFileSync(path.join(__dirname, 'index.html'));
    res.writeHead(200, { 'content-type': 'text/html; charset=utf-8' });
    return res.end(html);
  }
  if (req.method === 'GET' && url.pathname === '/events') {
    res.writeHead(200, { 'content-type': 'text/event-stream', 'cache-control': 'no-cache', connection: 'keep-alive' });
    res.write(`data: ${JSON.stringify(state.sessions)}\n\n`); // 全量快照
    clients.add(res);
    req.on('close', () => clients.delete(res));
    return;
  }
  if (req.method === 'POST' && url.pathname === '/hook') {
    let body = '';
    req.on('data', (c) => { body += c; if (body.length > 1e6) req.destroy(); });
    req.on('end', () => {
      try {
        const evt = JSON.parse(body || '{}');
        state = pruned(reduce(state, evt, Date.now()), Date.now());
        broadcast();
      } catch (_) {}
      res.writeHead(204); res.end();
    });
    return;
  }
  res.writeHead(404); res.end('not found');
});

server.listen(PORT, () => console.log(`拼豆 bridge listening http://localhost:${PORT}`));
```
