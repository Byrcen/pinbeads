```js
function drawPegboard(x,y,w,h,s){
  // 像素木框底板：方角 + 深木描边 + 上亮下暗 bevel（豆子/孔不在此函数，保持不动）
  const m=s*0.35, X=x-m, Y=y-m, W=w+2*m, H=h+2*m;
  const bw=Math.max(2,Math.round(s*0.12));
  bctx.fillStyle=TH.pegStroke; bctx.fillRect(X,Y,W,H);                    // 描边层
  bctx.fillStyle=TH.peg;       bctx.fillRect(X+bw,Y+bw,W-2*bw,H-2*bw);    // 板面
  bctx.fillStyle='rgba(255,246,220,.30)'; bctx.fillRect(X+bw,Y+bw,W-2*bw,bw);        // 顶部受光
  bctx.fillStyle='rgba(40,25,12,.22)';    bctx.fillRect(X+bw,Y+H-2*bw,W-2*bw,bw);    // 底部背光
  bctx.fillStyle=TH.pegDot;
  const p=state.pattern;
  for(let yy=0;yy<p.height;yy++)for(let xx=0;xx<p.width;xx++){
    bctx.beginPath(); bctx.arc(x+(xx+0.5)*s, y+(yy+0.5)*s, s*0.16, 0, 7); bctx.fill();
  }
}
```
